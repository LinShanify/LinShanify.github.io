Print Multiple Lines

print(str1, str2, sep=('\n'))

Format Output

# Python > 2.6
txt = "My name is {}, I'm {}".format("John", 36)
txt = "My name is {0}, I'm {1}".format("John", 36)
print("For only {price:.2f} dollars!".format(price=49))

# Python >= 3.6 (f-strings)
name = 'Tushar'
age = 23
print(f"Hello, My name is {name} and I'm {age} years old.")

Try / Except

try:
    print(x)
except:
    print("An exception occurred")

Raise Error

if not type(x) is int:
    raise TypeError("Only integers are allowed")

Check Value Type

# View type
type(i)

# Check type
isinstance(i, int)

# Hex string to int
int(x, 16)

String Operations

# str.upper()
"This is a 'string'.".upper()  # "THIS IS A 'STRING'."

# str.lower()
"This IS a 'string'.".lower()  # "this is a 'string'."

# str.capitalize()
"this Is A 'String'.".capitalize()  # "This is a 'string'."

# str.title()
"this Is a 'String'".title()  # "This Is A 'String'"

List Operations

a = [0, 1, 2, 3]

# Remove by value
a.remove(2)

# Remove by index
a.pop(1)

# Insert at index
a.insert(index, value)

# Apply function to all elements (Python 3: map returns iterator)
x = '1 2 3 4 5'
print(list(map(int, x.split(' '))))
